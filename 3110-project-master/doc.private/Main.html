val eval : env -> Ast.expr -> value

eval env e is the v such that <env, e> ==> v.

val eval_func_uniform : env -> Env.key -> Ast.num -> value

eval_func_uniform env id value is v such that <env, id(value)> ==> v with id the binding to an analytic function and value a Number.

val eval_func'_uniform : string -> Ast.f_expr -> Ast.f_expr -> float -> Ast.f_expr

eval_func'_uniform vs f acc gv is f' such that <_, f(vs) := vs(gv)...vs(gv)> ==> <_, f(vs) := gv...gv>.

val eval_root : value -> int -> value

eval_root f n is v such that <_,root(f)> ==> v.

val eval_func : env -> Env.key -> Ast.expr list -> value

eval_func id args is v such that <env, id(args)> ==> v with id the binding to an analytic function and args a one by n matrix.

val eval_func' : string -> Ast.f_expr -> mat -> Ast.f_expr -> Ast.f_expr

eval_func' vs f arr is f' such that <_, f(vs) := vs(arr.(0))...vs(arr.(n))> ==> <_, f(vs) := arr.(0)...arr.(n)>.

val eval_func'' : Ast.f_expr -> value
